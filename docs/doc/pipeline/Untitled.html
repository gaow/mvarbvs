filename = '/tmp/test.h5'

    if line[0] not in list(map(str, [x+1 for x in range(22)])):
        continue

                     index = [':'.join((x.split('_', 1)[0], y,z)) for x, y, z in zip(snps.snp, snps.a0, snps.a1)], 
                     columns = fam.iid, dtype = np.uint8)
    X.to_hdf(filename, 'chr{}/{}'.format(line[0], line[3]), mode = 'a', complevel = 9, complib = 'zlib')
    if(i%100 == 0):
        print('%.2f ' % ((float(i)/n)*100))
